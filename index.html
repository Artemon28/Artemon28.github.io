Сочи Автодром – единственный в России автодром, принимающий гонки Формулы 1. Он был спроектирован компанией Tilke Gmbh & Co. KG. и официально открыт в Олимпийском парке Сочи 20 сентября 2014 года. С 9 по 12 октября 2014 года на нем прошел первый в истории Гран-при России Формулы 1.Длина трассы Сочи Автодрома составляет 5848 метров. Она пролегает среди объектов, принимавших Зимние Олимпийские игры 2014. Всего на треке 18 поворотов: 12 правых и 6 левых. В конфигурации трассы обращает на себя внимание затяжной третий поворот — 750-метровый левый вираж с постоянным радиусом.

{
    long i, lk = 0, rk = 0;
    long j = 0;
    string L[100010], R[100010];
    for (i = p; i <= q; i++)
    {
        L[i - p].Country = a[i].Country;
        lk++;
        //cout << L[i-p] << " " << i-p<<"\n";
    }
    for (i = q + 1; i <= r; i++)
    {
        R[i - q - 1].Country = a[i].Country;
        rk++;
    }

    i = 0;
    for (int k = p; k <= r; k++)
    {
        if ((L[i].Country <= R[j].Country or j > rk-1) and i < lk)
        {
            a[k] = L[i];
            i++;
        }
        else {
            a[k] = R[j];
            j++;
        }
    }
    cout<<"\n";
}
